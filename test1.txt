#include <stdio.h>
#include <windows.h>
void main()
{
	SetConsoleCP(1251);
	SetConsoleOutputCP(1251);
	double R, h, S, V;
	const double pi = 3.14;
	printf("Введите значение радиуса ");
	scanf("%lf", &R);
	printf("Введите значение высоты ");
	scanf("%lf", &h);
	S = 2 * pi * R * R + 2 * pi * R * h;
	V = pi * R * R * h;
	printf("Площадь поверхности цилиндра = %.2lf\n", S);
	printf("Объем цилиндра = %.2lf\n", V);
}
