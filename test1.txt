#include <stdio.h> //подкл библиотеки вводы и вывода
#include <windows.h> //подкл библиотеки для кириллицы
void main() //заголовок функции
{/*Предусловие: радиус и высота цилиндра вводятся с консоли
	Постусловие: программа вычисляет площадь и объем цилиндра по заданным значениям R и h*/
	SetConsoleCP(1251);//Обращаемся к библиотеки для кириллицы
	SetConsoleOutputCP(1251);
	double R, h, S, V; //определяем вещественный тип значениям R, h, S, V
	const double pi = 3.14;//задаем константу (постоянное) значение для pi
	printf("Введите значение радиуса ");//выводим текст для пользователя
	scanf("%lf", &R); //вызывается консоль для задания (задаем) значения радиуса
	printf("Введите значение высоты ");//выводим текст для пользователя
	scanf("%lf", &h);//вызывается консоль для задания (задаем) значения высоты
	S = 2 * pi * R * R + 2 * pi * R * h; //формула площади поверхности цилиндра
	V = pi * R * R * h; //формула объема цилиндра
	printf("Площадь поверхности цилиндра = %.2lf\n", S);//выводим текст и значение S
	printf("Объем цилиндра = %.2lf\n", V);//выводим текст и значение V
}
